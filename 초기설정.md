# node.js 설치

# package.json 생성

npm init -y

# 의존성 패키지 설치

npm install express cors esbuild body-parser typescript three cannon-es

npm install @types/three

npm install --save-dev nodemon

# package.json 파일을 열어서 "type": "commonjs", 를 다음과 같이 수정

"type": "module",

# package.json 파일을 열어서 scripts 섹션을 다음과 같이 수정

"scripts": {
"start": "node server.js",
"dev": "nodemon server.js"
}

# 서버 시작

npm start
